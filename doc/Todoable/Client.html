def authenticate
  @options[:basic_auth] = {
    username: @username,
    password: @password
  }
  response = self.class.post('/authenticate', @options)
  check_and_raise_errors(response)
  @token = response.parsed_response['token']
  # tokens expire after 20 minutes
  @token_expiry = Time.now + (20 * 60)
  self
end
